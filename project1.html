Step 1 — Load & Clean the Data

The raw dataset contained duplicates, missing conversion flags, and inconsistent discount formatting. Below is the cleaning pipeline.

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from matplotlib.ticker import FuncFormatter

# Load dataset
df = pd.read_csv('promotions_data.csv')
print(f"Raw rows: {len(df)}")  # 2,201

# Drop duplicates and rows with missing key fields
df = df.drop_duplicates()
df = df.dropna(subset=['discount_pct', 'converted', 'revenue'])
print(f"Clean rows: {len(df)}")  # 2,147

# Normalize discount into tiers
def assign_tier(pct):
    if pct <= 12: return 'Low (5–12%)'
    elif pct <= 25: return 'Mid (13–25%)'
    else: return 'High (26–50%)'

df['tier'] = df['discount_pct'].apply(assign_tier)
df['revenue'] = pd.to_numeric(df['revenue'], errors='coerce').fillna(0)

# Flag converted transactions only
df_conv = df[df['converted'] == 1].copy()

Grouped metrics by tier to compare conversion rates, average revenue per transaction, and total revenue contribution.

Step 4 — Automate the Pipeline

Wrapped all steps into a reusable function so the full report regenerates with one call on any new dataset.

def run_promo_analysis(filepath: str) -> pd.DataFrame:
    """
    Full pipeline: load → clean → tier → aggregate → export report.
    Returns tier_stats DataFrame. Saves chart to /reports/.
    """
    df = pd.read_csv(filepath).drop_duplicates()
    df = df.dropna(subset=['discount_pct', 'converted', 'revenue'])
    df['tier'] = df['discount_pct'].apply(assign_tier)
    df['revenue'] = pd.to_numeric(df['revenue'], errors='coerce').fillna(0)

    stats = df.groupby('tier').agg(
        total=('converted', 'count'),
        converted=('converted', 'sum'),
        avg_revenue=('revenue', 'mean'),
        total_rev=('revenue', 'sum')
    ).reset_index()
    stats['conv_rate'] = (stats['converted'] / stats['total'] * 100).round(2)

    _build_charts(stats)
    return stats

# One-line execution on any new file
results = run_promo_analysis('promotions_q4.csv')
print(results)
